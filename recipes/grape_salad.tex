\recipe[Who the heck is Donna?]{Donna's Grape Salad}{Gran}
\serves{8}
\preptime{10 minutes}
% \cooktime{}
\dishtype{}
\begin{ingreds}
    4 cups grapes
    1 cup nuts (pecans)
    1 cup sour cream
    1 cup brown sugar
    1$\frac{1}{2}$ tsp almond extract
\end{ingreds}
\begin{method}
    Mix the sour cream, brown sugar, and almond extract.\par
    Pour the mixture over the grapes and nuts.\par
    Mix until completely covered.\par
    Chill and serve.
\end{method}
