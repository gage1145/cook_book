\recipe[Made at every family get-together.]{Guacamole}{Boss}
\serves{4}
\preptime{15 minutes}
% \cooktime[Cook time]{0 minutes}
\dishtype{\apps}
\dishother{\vegetarian}
\begin{ingreds}
    4 Avocados
    $\frac{1}{3}$ onion
    $\frac{1}{2}$ jalapeno
    $\frac{1}{4}$ cup cilantro
    1 tbsp mayo
    2 tsp vinegar
    $\frac{1}{2}$ lemon
    garlic salt
    pepper
    tomatoes (optional)
    stuffed olives (optional)
    cucumbers (optional)
\end{ingreds}
\begin{method}
    Mix and mash all the ingredients together.
\end{method}
