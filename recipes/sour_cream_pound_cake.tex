\recipe[Dense and melt-in-your-mouth.]{Sour Cream Pound Cake}{Mimi}
\serves{8--10}
\preptime{10 minutes}
\cooktime{1 hour}
\dishtype{\dessert}
\dishother{\vegetarian}
\begin{ingreds}
    3 cups sugar
    6 eggs
    2 sticks butter
    1 cup sour cream
    3 cups flour
    $\frac{1}{4}$ tsp baking powder
    1$\frac{1}{2}$ tsp vanilla extract
\end{ingreds}
\begin{method}[Preheat the oven to \temp{325}.]
    Mix all the ingredients together.\par
    Add mixture to a baking pan.\par
    Bake for 1 hour.
\end{method}
