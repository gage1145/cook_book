\recipe[You actually don't have to be poor to make this.]{Poverty Meatloaf}{Gran}
\serves{8}
\preptime{20 minutes}
\cooktime{1$\frac{1}{2}$ hours}
\dishtype{\main}
\dishother{}
\begin{ingreds}
    2 lbs ground beef
    bread (chopped)
    1 onion (diced)
    1 can tomatoes
    1 eggs
    garlic powder
    salt
    black pepper
    ketchup
\end{ingreds}
\begin{method}[Preheat the oven to \temp{350}.]
    Mix all the ingredients except for the ketchup together until homogeneous.\par
    Top with the ketchup.\par
    Bake for 1$\frac{1}{2}$ hours.
\end{method}
