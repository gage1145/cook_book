\recipe[A warm, vegetarian option for the winter.]{Broccolli Casserole}{Gran}
\serves{8}
\preptime{10 minutes}
\cooktime{1$\frac{1}{4}$ hours}
\dishtype{\vegetarian}
\begin{ingreds}
    1 lb frozen broccoli
    $\frac{1}{2}$ cup butter
    1 can cream of mushroom soup
    1 can cream of chicken soup
    1 jar of Cheez Whiz\textsuperscript{\textregistered}
    3--4 cups cooked rice
\end{ingreds}
\begin{method}[Preheat the oven to \temp{350}.]
    In a 6 quart casserole dish, melt the butter and cook the frozen broccoli until done.\par
    Add the rest of the ingredients and mix thoroughly.\par
    Bake for 1 hour.
\end{method}
