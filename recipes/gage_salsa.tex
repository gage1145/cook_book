\recipe[A family favorite enjoyed by those that like spice.]{Salsa}{Gage}
\serves{4}
\preptime{10 minutes}
% \cooktime[Cook time]{0 minutes}
\dishtype{\apps}
\dishother{\vegetarian}
\begin{ingreds}
    1 can diced tomatoes
    $\frac{1}{2}$ onion
    3 jalape\~nos
    1 bunch cilantro
    1 lemon (juice)
    2 tsp garlic salt
    1 tsp cumin
    $\frac{1}{2}$ tsp cayenne pepper
    back pepper (to taste)
\end{ingreds}
\begin{method}
    Cut the onions and jalape\~nos into large chunks and place into a food processor.\par
    Add the rest of the ingredients.\par
    Blend to desired consistency.
\end{method}
