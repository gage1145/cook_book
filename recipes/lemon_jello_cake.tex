\recipe[Honestly this one also feels like a Depression-era recipe.]{Lemon Jello Cake}{Gran}
\serves{8--10}
\preptime{10 minutes}
\cooktime{45 minutes}
\dishtype{}
\begin{ingreds}
    \ingredients[Cake]
        4 eggs
        1 yellow cake Mix
        1 pkg lemon Jello
        1 cup water
        $\frac{1}{2}$ cup vegetable oil
        2 tsp lemon extract
    \columnbreak{}
    \ingredients[Glaze]
        1 cup powdered sugar
        $\frac{1}{2}$ cup lemon juice
        zest of 1 lemon
\end{ingreds}
\begin{method}[Preheat the oven to \temp{325}.]
    Mix all the cake ingredients together until homogeneous.\par
    Add the mixture to a floured tube pan.\par
    Bake for 45 minutes.\par
    Meanwhile, mix the glaze ingredients together until homogeneous.\par
    Pour generously over the cake once done baking.
\end{method}
